{{/* Seeking until there are no children, recursively. */}} {{- $currentPage := . -}} {{- $Result := false -}} {{- $children := $currentPage.Pages.ByWeight -}} {{- $lastChild := index ( last 1 $children ) 0 -}} {{- if ge $lastChild.Pages.Len 1 -}} {{- $Result = ( partial "navigation/section_traversal/prev_recursive.html" $lastChild ) -}} {{- else -}} {{- $Result = $lastChild -}} {{- end -}} {{- return $Result -}}
